// ConsoleApplication24.cpp : 定义控制台应用程序的入口点。
//

#include "stdafx.h"
#include"stdio.h"


int main()
{
	int i, j, m, n;
	float a[3][100], w = 0, t, u = 0;
	scanf_s("%d %d", &m, &n);//输入月饼种类和市场需求
	for (i = 0; i < 2; i++)
		for (j = 0; j < m; j++)
			scanf_s("%d", &a[i][j]);

	for (i = 0; i < m; i++)
		a[2][i] = a[1][i] / a[0][i];//月饼单价

	for (j = 0; j < m - 1; j++)
		for (i = 0; i < m - 1 - j; i++)
			if (a[2][i] < a[2][i + 1])
			{
				t = a[2][i];
				a[2][i] = a[2][i + 1];
				a[2][i + 1] = t;
				t = a[1][i];
				a[1][i] = a[1][i + 1];
				a[1][i + 1] = t;
				t = a[0][i];
				a[0][i] = a[0][i + 1];
				a[0][i + 1] = t;
			}//对月饼的单价进行冒泡排序
	i = 0;
	while (u < n)
	{
		w += a[1][i];
		u += a[0][i];
		i++;
	}

	w = w - (u - n)*a[2][i - 1];
	printf("%.2f", w);

	return 0;


}
