

#include"stdio.h"

void Insertsort(int *num, int n)
{
	int i = 0;
	int j = 0;
	int tmp = 0;
	for (i = 1; i < n; i++)
	{
		tmp = num[i];
		j = i - 1;
		while (j >= 0 && tmp < num[j])
		{
			num[j + 1] = num[j];
			j--;
		}
		num[j + 1] = tmp;
	}
}

int main()
{
	int i;
	int a[8] = { 3,5,2,8,1,7,9,0 };
	Insertsort(a, 8);
	for (i = 0; i < 8; i++)
		printf("%d",a[i])
	
    return 0;
}
